  <h2>Chat Log</h2>
  <div id="chat"></div>

  <script>
    const ws = new WebSocket('ws://localhost:3004/ws');
    const chat = document.getElementById('chat');
    const myId = document.getElementById('myId');

    ws.onopen = () => {
      console.log('âœ… Connected to WS');
    };

    ws.onmessage = (e) => {
      const data = JSON.parse(e.data);
      console.log(data);

      if (data.type === 'id') {
        myId.textContent = data.id;
      } else if (data.from && data.msg) {
        chat.innerHTML += `<div><strong>${data.from}:</strong> ${data.msg}</div>`;
      } else if (data.invite) {
        chat.innerHTML += `<div style="color: blue;">
          Invite from ${data.from}: ${data.invite}
          <button onclick="acceptInvite('${data.from}')">Accept</button>
          <button onclick="declineInvite('${data.from}')">Decline</button>
        </div>`;
      } else if (data.accept) {
        chat.innerHTML += `<div style="color: green;"><strong>${data.from} accepted your invite!</strong></div>`;
      } else if (data.decline) {
        chat.innerHTML += `<div style="color: red;"><strong>${data.from} declined your invite.</strong></div>`;
      } else if (data.profileOf) {
        chat.innerHTML += `<div><strong>Profile Link:</strong> <a href="${data.link}" target="_blank">${data.profileOf}</a></div>`;
      }
    };

    document.getElementById('send').onclick = () => {
      const to = document.getElementById('to').value;
      const msg = document.getElementById('msg').value;

      ws.send(JSON.stringify({
        type: 'msg',
        to,
        msg
      }));

      chat.innerHTML += `<div style="color: gray;">You to ${to}: ${msg}</div>`;
    };

    document.getElementById('block').onclick = () => {
      const blockId = document.getElementById('blockId').value;

      ws.send(JSON.stringify({
        type: 'block',
        block: blockId
      }));

      chat.innerHTML += `<div style="color: red;">You blocked ${blockId}</div>`;
    };

    document.getElementById('invite').onclick = () => {
      const inviteId = document.getElementById('inviteId').value;

      ws.send(JSON.stringify({
        type: 'invite',
        to: inviteId
      }));

      chat.innerHTML += `<div style="color: blue;">You invited ${inviteId} to Pong</div>`;
    };

    document.getElementById('profile').onclick = () => {
      const profileId = document.getElementById('profileId').value;

      ws.send(JSON.stringify({
        type: 'profile',
        user: profileId
      }));

      chat.innerHTML += `<div style="color: purple;">Requested profile of ${profileId}</div>`;
    };

    function acceptInvite(from) {
      ws.send(JSON.stringify({ type: "accept", to: from }));
      chat.innerHTML += `<div style="color: green;">You accepted invite from ${from}</div>`;
    }

    function declineInvite(from) {
      ws.send(JSON.stringify({ type: "decline", to: from }));
      chat.innerHTML += `<div style="color: red;">You declined invite from ${from}</div>`;
    }
  </script>
